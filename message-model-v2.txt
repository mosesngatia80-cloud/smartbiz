const mongoose = require("mongoose");

const MessageSchema = new mongoose.Schema({
  orderId: {
    type: String
  },

  businessId: {
    type: String
  },

  sender: {
    type: String,
    required: true
  },

  message: {
    type: String,
    required: true
  }

}, {
  timestamps: true
});

module.exports = mongoose.model(
  "Message",
  MessageSchema
);
