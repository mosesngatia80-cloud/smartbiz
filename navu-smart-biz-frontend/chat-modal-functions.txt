function closeChatModal() {

  document.getElementById(
    "chatModal"
  ).style.display = "none";
}

function sendCurrentChatMessage() {

  const input =
    document.getElementById(
      "chatInput"
    );

  const message =
    input.value.trim();

  if (!message) return;

  sendChatMessage(message);

  input.value = "";
}
