REPLACE WITH THIS:

await order.save();

// 💰 HANDLE WALLET PAYMENT
const payment = await handleWalletPayment(order);

if (!payment.paid) {
  return res.json({
    success: true,
    order,
    payment: {
      status: "FAILED",
      reason: payment.reason
    }
  });
}

console.log("🛒 ORDER + PAYMENT SUCCESS:", order._id.toString());

return res.json({
  success: true,
  order,
  payment: {
    status: "PAID",
    remainingBalance: payment.balance
  }
});
